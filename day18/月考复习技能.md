     var mybs = new BScroll('.main', {
       click:true,
       probeType: 2,
       pullDownRefresh: {
         threshold: 180,
         stop: 90
       },
       pullUpLoad: {
         threshold: 180
       }
     }) 

     mybs.on('pullingDown', function() {
        setTimeout(function() {
          getData((res) => {
            render(res, 'top')
            mybs.finishPullDown()
            mybs.refresh()
          })  
        }, 2000)  
     })

      mybs.on('pullingUp', function() {
        setTimeout(function() {
          getData((res) => {
            render(res, 'bottom')
            mybs.finishPullUp()
            mybs.refresh()
          })  
        }, 2000)  
     })

# 排序

function sortFn(arr, msg) {
  arr.sort((a, b) => {
    if(a[msg] > b[msg]) {
      return 1
    } else if (a[msg] < b[msg]) {
      return -1
    } else {
      return 0;
    } 
  })
  <!-- render(arr); -->
  return arr;
}
